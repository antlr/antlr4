[grammar]
grammar G;

root
    : {0==0}? continue+ {<writeln("$text")>}
    ;

continue returns [int return]
    : for for? {1==1}?              #else
    | break=BREAK BREAK+ (for | IF) #class
    | if+=IF  if+=IF*               #int
    | continue CONTINUE             #static
    ;

for: FOR;
FOR: 'for_';
BREAK: 'break_';
IF: 'if_';
CONTINUE: 'continue_';
