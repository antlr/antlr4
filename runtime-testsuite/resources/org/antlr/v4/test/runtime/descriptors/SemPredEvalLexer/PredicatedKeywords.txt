[type]
Lexer

[grammar]
lexer grammar L;
ENUM : [a-z]+ { <TextEquals("enum")> }? { <writeln("\"enum!\"")> } ;
ID   : [a-z]+ { <PlusText("ID "):writeln()> } ;
WS   : [ \n] -> skip ;

[input]
enum enu a

[output]
enum!
ID enu
ID a
[@0,0:3='enum',<1>,1:0]
[@1,5:7='enu',<2>,1:5]
[@2,9:9='a',<2>,1:9]
[@3,10:9='<EOF>',<-1>,1:10]
