[type]
Lexer

[grammar]

ID : [a-zA-Z] [a-zA-Z0-9]* {<writeln("\"ID\"")>} ;
WS : [ \n\\u0009\r]+ -> skip ;

[input]
"""34
 34 a2 abc 
   """

[output]
I
I
ID
ID
[@0,0:1='34',<1>,1:0]
[@1,4:5='34',<1>,2:1]
[@2,7:8='a2',<2>,2:4]
[@3,10:12='abc',<2>,2:7]
[@4,18:17='<EOF>',<-1>,3:3]
