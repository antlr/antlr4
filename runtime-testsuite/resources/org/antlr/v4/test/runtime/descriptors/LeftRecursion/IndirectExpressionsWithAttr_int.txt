[grammar]
grammar T;
s @after {<ToStringTree("$ctx"):writeln()>} : e EOF ; // must indicate EOF can follow
e : e '.' ID
  | b[0]
  | m[0]
  | e '*' e
  | e ('+'|'-') e
  | INT
  | ID
  ;
b [int x]: '(' e ')';
m [int y] : '-' e;
ID : 'a'..'z'+ ;
INT : '0'..'9'+ ;
WS : (' '|'\n') -> skip ;

[start]
s

-(-a+b)+c

"""
